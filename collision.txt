static EVENT_HANDLER(collision) {
  fprintf(stdout, "collision on frame %d\t", (int)frame_length);
  srand(time(NULL));
  int c;
  if(tempframe.collisions == 16) {
    exit(EXIT_FAILURE);       // Report ethernet severed to network layer
  }
  else if(tempframe.collisions >= 10) {
    c = rand() % (int)pow(2, 10);   // Back off for maximum time for 6 more collisions
  }
  else if(tempframe.collisions < 10) {
    c = rand() % (int)pow(2, tempframe.collisions);        // Calculate backoff time
  }
  CnetTime backoff = ((CnetTime)51.2 * c);
  lasttimer = CNET_start_timer(EV_TIMER2, backoff, 0);                                         
}
